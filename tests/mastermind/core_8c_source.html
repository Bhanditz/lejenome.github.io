318  session->state == MM_SUCCESS)
319  return 1;
320  mm_guess *G = session->panel + session->guessed;
321  uint8_t *freq =

323  for (i = 0; i < session->config->colors; i++)
324  freq[i] = 0;

331  free(freq);
332  return 1;
333  }

336  G->insecret++;

342  session->state = MM_SUCCESS;

344  session->state = MM_FAIL;

358 long unsigned mm_score(mm_session *session)
359 {
360  unsigned i;
361  if (session->state == MM_FAIL)
362  return 0;
363  // Max score it can get on current session config
364  long unsigned score = session->config->holes * session->config->colors *
365  MM_GUESSES_MAX * 2;
366  // for each played guess, 1 time minus on secret but on wrong place and
367  // 2 time minus not on scecret at all
368  for (i = 0; i < session->guessed; i++)
369  score -=
370  (session->config->holes * 2) -
371  (session->panel[i].inplace + session->panel[i].insecret);
372  // -%25 for remise mode
373  if (session->config->remise)
374  score *= 0.75;
375  return score;
376 }
