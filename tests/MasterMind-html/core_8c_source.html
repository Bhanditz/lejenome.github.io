327 unsigned mm_play(mm_session *session, uint8_t *T)
328 {
329  char i;
330  if (session->guessed >= session->config->guesses ||
331  session->state == MM_SUCCESS)
332  return 1;
333  mm_guess *G = session->panel + session->guessed;
334  uint8_t *freq =
335  (uint8_t *)malloc(sizeof(uint8_t) * session->config->colors);
336  for (i = 0; i < session->config->colors; i++)
337  freq[i] = 0;
338  G->combination = T;
339  G->right_pos = 0;
340  G->wrong_pos = 0;
341  for (i = 0; i < session->config->holes; i++) {
342  if (T[i] >= session->config->colors ||
343  (session->config->remise && freq[T[i]] > 0)) {
344  free(freq);
345  return 1;
346  }
347  freq[T[i]]++;
348  if (freq[T[i]] <= session->secret->freq[T[i]])
349  G->wrong_pos++;
350  if (session->secret->val[i] == T[i])
351  G->right_pos++;
352  }
353  G->wrong_pos -= G->right_pos;
354  session->guessed++;
355  // update session state
356  if (G->right_pos == session->config->holes)
357  session->state = MM_SUCCESS;
358  else if (session->guessed == session->config->guesses)
359  session->state = MM_FAIL;
360  else
361  session->state = MM_PLAYING;
362  free(freq);
363  // save session if save_on_play option is enabled
364  if (mm_confs[MM_POS_SAVE_PLAY].bool.val == 1)
365  mm_session_save(session);
366  // save score on success
367  if (session->state == MM_SUCCESS)
368  mm_scores_save(session);
369  return 0;

375 long unsigned mm_score(mm_session *session)
376 {
377  unsigned i;
378  if (session->state == MM_FAIL)
379  return 0;
380  // Max score it can get on current session config
381  long unsigned score = session->config->holes * session->config->colors *
382  MM_GUESSES_MAX * 2;
383  // for each played guess, 1 time minus on secret but on wrong place and
384  // 2 time minus not on scecret at all
385  for (i = 0; i < session->guessed; i++)
386  score -= (session->config->holes * 2) -
387  (session->panel[i].right_pos * 2 +
388  session->panel[i].wrong_pos);
389  // -%25 for remise mode
390  if (session->config->remise)
391  score *= 0.75;
392  return score;
393 }
